            fact ,copy = 1,num

                fact *= copy
                copy -= 1
            return f"{num} Factorial is = {fact}"

        def check_armstrong(num):
            copy,sum = num,0

            while copy != 0:
                rem = copy % 10 
                sum += rem ** len(num)        
                copy = int(copy / 10)
                
            return f"{num} is Armstrong Number\n" if num == sum else f"{num} is Not Armstrong Number\n"
